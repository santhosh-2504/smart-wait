Basic Usage
import { wait } from 'smart-wait';

// Wait for 2 seconds
await wait(2000);
console.log('2 seconds later!');
Cancellable Timeout
const { id, promise } = waitWithId(5000);

// Cancel after 2 seconds
setTimeout(() => cancelWait(id, 'User cancelled'), 2000);

try {
    await promise;
} catch (err) {
    console.error(err.message); // "User cancelled"
}
